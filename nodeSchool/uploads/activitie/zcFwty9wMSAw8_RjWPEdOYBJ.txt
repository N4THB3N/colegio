function updateTeacher(req, res){
  var teacherId = req.params.id;
  var update = req.body;

  if(teacherId != req.admin.sub){
    res.status(500).send({
      message: 'Incorrect token or id entered'
    })
  }

  Teacher.findByIdAndUpdate(teacherId, update, {new: true}, (err, teacherUpdate) => {
    if(err){
      res.status(500).send({
        message: 'There was an error while updating the teacher'
      });
    }else{
      if(!teacherUpdate){
        res.status(404).send({
          message: 'Unable to update the record from teacher collection'
        });
      }else{
        res.status(200).send({
          teacher: teacherUpdate
        });
      }
    }
  });
}
